138#define foreach_helper_init( SIZE, ITEM_NAME ) \
139 for ( size_t foreach_keep_##ITEM_NAME = 1, foreach_index_##ITEM_NAME = 0, \
140 foreach_cap_##ITEM_NAME = ( SIZE ); \
141 foreach_keep_##ITEM_NAME \
142 && foreach_index_##ITEM_NAME < foreach_cap_##ITEM_NAME; \
143 foreach_keep_##ITEM_NAME = !foreach_keep_##ITEM_NAME, \
